
# Product Requirements Prompt (PRP) — 🔘 Pending

## 1. Project Context & Objectives

The previous `MediaStream` approach failed because the `standardized-audio-context` wrapper used by Tone.js does not implement `createMediaStreamDestination()`. This resulted in a silent catch-block failure, dropping the entire generative sequence to the dry master output.

**Objective:** Implement Module-Level Context Injection. We will force Tone.js to adopt a pure native `AudioContext` the millisecond the module loads, ensuring the `Tone.Transport` singleton binds correctly to the Web Worker clock. This allows us to cleanly bridge Tone nodes to the Worklet node using standard native routing without triggering `InvalidAccessError` or `TypeError`.

## 2. Technical Decisions & Dependencies

* **Module-Level Injection:** `Tone.setContext()` will be executed globally at the top of `GenerativeAudioController.js` rather than imperatively inside a React component.
* **Routing Simplification:** All `MediaStream` hackery will be removed. We will return to `Tone.connect()` and standard native Web Audio bridging.

## 3. Implementation Phases

### Phase 1: Module-Level Context Injection (GenerativeAudioController.js) — ✅ Completed

* **Objective:** Bind the native context before the `GenerativeAudioController` class is instantiated.
* **Instructions:**

```javascript
// INJECT NATIVE CONTEXT IMMEDIATELY TO ALIGN SINGLETONS
// This prevents Tone.Transport clock freezing and bypasses the polyfill InvalidAccessErrors.
if (typeof window !== 'undefined') {
  const NativeAudioContext = window.AudioContext || window.webkitAudioContext;
  if (NativeAudioContext && !window.somnusNativeContextInjected) {
    const nativeCtx = new NativeAudioContext(); // Suspended by default
    Tone.setContext(nativeCtx);
    window.somnusNativeContextInjected = true;
  }
}

```

```javascript
async bootEngine() {
  if (this.isInitialized) return;

  // 1. Instantiate Tone buses
  this.masterBus = new Tone.Volume(-6).toDestination();
  this.reverbInputBus = new Tone.Gain(1);

  // 2. Extract the pure native context (Guaranteed by module injection above)
  const nativeContext = Tone.getContext().rawContext;

  // 3. Initialize Native AudioWorklet & Route
  try {
    if (nativeContext && nativeContext.audioWorklet) {
      await nativeContext.audioWorklet.addModule('/assets/fdn-reverb-worklet.js');
      this.reverbWorklet = new window.AudioWorkletNode(nativeContext, 'fdn-reverb-worklet', {
        numberOfInputs: 1,
        numberOfOutputs: 1,
        outputChannelCount: [2],
      });
      this.reverbNode = this.reverbWorklet;

      // Bridge Tone -> Worklet
      Tone.connect(this.reverbInputBus, this.reverbWorklet);

      // Bridge Worklet -> Native Destination
      const nativeGain = nativeContext.createGain();
      nativeGain.gain.value = 0.5; // -6dB headroom
      this.reverbWorklet.connect(nativeGain);
      nativeGain.connect(nativeContext.destination);
    } else {
      throw new Error("Worklet not supported on this context.");
    }
  } catch (e) {
    console.warn('Worklet load fallback/test environment:', e);
    this.reverbWorklet = null;
    // Fallback routing
    this.reverbInputBus.connect(this.masterBus);
  }

  // 4. Re-instantiate Synths
  try {
    this.layer1Synth = new Tone.PolySynth(Tone.Synth, {
      oscillator: { type: 'sine' },
      envelope: { attack: 4, decay: 2, sustain: 1, release: 4 },
    }).connect(this.reverbInputBus);
    this.layer1Synth.volume.value = -12;

    this.layer2Filter = new Tone.Filter({
      frequency: 2000, type: 'lowpass', rolloff: -12,
    }).connect(this.reverbInputBus);

    this.layer2Synth = new Tone.PolySynth(Tone.Synth, {
      oscillator: { type: 'triangle' },
      envelope: { attack: 2, decay: 1, sustain: 0.8, release: 3 },
    }).connect(this.layer2Filter);
    this.layer2Synth.volume.value = -14;

    this.layer3Filter = new Tone.Filter({
      frequency: 3000, type: 'lowpass', rolloff: -12,
    }).connect(this.reverbInputBus);

    this.layer3Synth = new Tone.PolySynth(Tone.Synth, {
      oscillator: { type: 'sine' },
      envelope: { attack: 1.5, decay: 1, sustain: 0.7, release: 2.5 },
    }).connect(this.layer3Filter);
    this.layer3Synth.volume.value = -16;
  } catch (e) {
    console.warn('Synth instantiation fallback:', e);
  }

  this.isInitialized = true;
}

```
